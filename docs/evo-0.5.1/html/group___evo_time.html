typedef TimerT<TimerStampWall> Timer

Timer that works like a stopwatch and measures real (wall clock) time.

  • See TimerT for full interface
Example
#include <evo/timer.h>
#include <evo/io.h>
using namespace evo;
int main() {
Timer timer;
timer.start();
sleepus(1);
timer.stop();
con().out << "Slept for " << timer.nsec() << " nsec" << NL;
return 0;
}
