1153  state.current = NULL;
1154  state.prev_arg = NULL;
1155  return true;
1156  } else if (arg == "--") {
1157  // Stop processing options
1158  state.end_options = true;
1159  state.current = NULL;
1160  return true;
1161  }
1162  }
1163 
1164  if (state.current == NULL && state.prev_arg != NULL) {
1165  // Switch back to previous argument with multiple values
1166  state.current = state.prev_arg;
1167  state.valnum = 1;
1168  }
1169 
1170  if (state.current != NULL) {
1171  if (!state.end_options && arg.starts('-') && state.valnum > 0) {
1172  // Found an option between arguments
1173  if (state.current->type == tPOS && state.current->bitflags & BITFLAG_MULTI_VALUE)
1174  state.prev_arg = state.current; // save multi-value argument to come back to
1175  state.current = NULL;
1176  state.valnum = 0;
1177  } else {
1178  // Value for current option/argument
1179  const OptionInfo& info = *state.current;
1180  if (!parse_validate(info, arg))
1181  return false;
1182  info.store_value(map, arg);
1183  state.value_added(info);
1184  return true;
1185  }
1186  }
1187 
1188  if (!state.end_options) {
1189  // Check for options
1190  SubString str, name, value;
1191  if (arg.starts("--", 2)) {
1192  // Long option
1193  str.set(arg, 2);
1194  const bool found_value = str.split('=', name, value);
1195 
1196  const OptionInfo* infop = option_lookup(state, name);
1197  if (infop == NULL) {
1198  if (name.starts("no-", 3)) {
1199  // Check for --no-* option to reset (remove) option
1200  SubString noname(name, 3);
1201  infop = option_lookup(state, noname);
1202  if (infop != NULL) {
1203  if (found_value) {
1204  con_.err << progname_ << ": ERROR: Value not allowed with reset option: --" << name << newline_;
1205  return finish(true);
1206  }
1207  map.remove(infop->key);
1208  return true;
1209  }
1210  }
1211  con_.err << progname_ << ": ERROR: Unknown option: --" << name << newline_;
1212  return finish(true);
1213  }
1214 
1215  const OptionInfo& info = *infop;
1216  if (info.type == tFLAG) {
1217  if (found_value) {
1218  con_.err << progname_ << ": ERROR: Unexpected value with option: --" << name << newline_;
1219  return finish(true);
1220  }
1221  info.store_flag(map);
1222  } else if (found_value) {
1223  if (!parse_validate(info, value))
1224  return false;
1225  info.store_value(map, value);
1226  state.value_stored(info);
1227  } else
1228  state.value_expected(info);
1229  return true;
1230  } else if (arg.starts('-')) {
1231  // Short options
1232  const SubString args(arg.data() + 1, arg.size() - 1);
1233  for (StrSizeT i = 0, c = args.size(); i < c; ++i) {
1234  if (args[i] == 'h') {
1235  print_help(state.cur_cmd);
1236  return finish(false);
1237  } else {
1238  str.set(args.data() + i, 1);
1239  const OptionInfo* infop = option_lookup(state, str);
1240  if (infop == NULL) {
1241  con_.err << progname_ << ": ERROR: Unknown option: -" << str << newline_;
1242  return finish(true);
1243  }
1244 
1245  const OptionInfo& info = *infop;
1246  if (info.type == tFLAG) {
1247  if (i + 1 < c && args[i + 1] == '=') {
1248  con_.err << progname_ << ": ERROR: Unexpected value with option: -" << args[i] << newline_;
1249  return finish(true);
1250  }
1251  info.store_flag(map);
1252  } else {
1253  if (i + 1 >= c) {
1254  state.value_expected(info);
1255  return true; // Value expected in next argument
1256  }
1257  if (args[i + 1] == '=') {
1258  ++i;
1259  } else if (i > 0) {
1260  str.set(args.data() + i + 1, c - i - 1);
1261  con_.err << progname_ << ": ERROR: Possible typo, use '" << args[i] << '=' << str << "' for clarity when combining short options in: -" << args << newline_;
1262  return finish(true);
1263  }
1264 
1265  str.set(args.data() + i + 1, c - i - 1);
1266  if (!parse_validate(info, str))
1267  return false;
1268  info.store_value(map, str);
1269  state.value_stored(info);
1270  break;
1271  }
1272  }
1273  }
1274  return true;
1275  }
1276  }
1277 
1278  // Argument value
1279  if (state.argnum >= state.cur_args->size()) {
1280  if (subcommands_.size() > 0 && state.cur_cmd == NULL) {
1281  // Sub-command argument
1282  state.cur_cmd = subcommands_.find(arg);
1283  if (state.cur_cmd == NULL) {
1284  con_.err << progname_ << ": ERROR: Unknown command: " << arg << newline_;
1285  return finish(true);
1286  }
1287  map["command"] = arg;
1288  state.current = NULL;
1289  state.prev_arg = NULL;
1290  state.cur_args = &(state.cur_cmd->args);
1291  state.argnum = 0;
1292  state.valnum = 0;
1293  return true;
1294  }
1295 
1296  con_.err << progname_ << ": ERROR: Unexpected argument: " << arg << newline_;
1297  return finish(true);
1298  }
1299 
1300  const OptionInfo& info = (*state.cur_args)[state.argnum];
1301  if (!parse_validate(info, arg))
1302  return false;
1303  info.store_value(map, arg);
1304  state.value_stored(info);
1305  ++state.argnum;
1306  return true;
1307  }
