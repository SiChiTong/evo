1386  if (count <= 0)
1387  return 0;
1388  else if (start <= 0)
1389  return -(lbit >> (count-1));
1390  else
1391  return ( ~-(lbit >> (start-1)) & -(lbit >> (start+count-1)) );

1396  data = 0;
1397  if (count > 0) {
1398  ptr = &(ptr[offset / bits]);
1399  offset = offset % bits;
1400  if (offset + count <= bits)
1401  // Copy from single byte
1402  data = (*ptr & maskM(offset, count)) >> (bits - (offset+count));
1403  else {
1404  // Copy from more than one byte
1405  if (count > BitInfoT<T>::bits)
1406  count = BitInfoT<T>::bits;
1407  // Copy first partial byte
1408  if (offset > 0) {
1409  data |= ( (*ptr & ~-(lbit >> (offset-1))) << (count-offset) );
1410  count -= (bits - offset);
1411  ++ptr;
1412  }
1413  // Copy whole bytes
1414  while (count >= bits) {
1415  count -= bits;

1418  }
1419  // Copy end partial byte
1420  if (count > 0)
1421  data |= ( (*ptr & -(lbit >> (count-1))) >> (bits-count) );
