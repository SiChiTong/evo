743  // Search "others" for key, using compare, set index (insertion point if not found)
744  const Item* search(Size& index, const Key& key, const Compare& compare) const {
745  int cmp;
746  Size left = 0, right = others.size(), mid = 0;
747  const Item* items = others.data();
748  while (left < right) {
749  mid = left + ((right-left) / 2);
750  const Item* item = items + mid;
751  cmp = compare(key, item->key);
752  if (cmp < 0) {
753  right = mid;
754  } else if (cmp == 0) {
755  index = mid;
756  return item;
757  } else
758  left = mid + 1;
759  }
760  index = left;
761  return NULL;
762  }
