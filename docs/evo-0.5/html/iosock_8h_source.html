213  bool parse(const SubString& str, ushort port=0, int family=AF_UNSPEC) {
214  clear();
215  if (str.size() == 0)
216  return false;
217  const char* pstr = str.data();
218  const char* pend = pstr + str.size();
219 
220  if (family == AF_UNSPEC) {
221  // Auto-detect: Check first few chars for IPv4 syntax, fallback to IPv6
222  int i = 0;
223  for (const char* p=pstr; p < pend; ++p, ++i) {
224  if (*p == '.') {
225  --i;
226  break;
227  }
228  if (*p < '0' || *p > '9') {
229  i = -1;
230  break;
231  }
232  }
233  family = (i < 0 ? AF_INET6 : AF_INET);
234  }
235 
236  switch (family) {
237  case AF_INET: {
238  if (str.size() >= MAX_INET4_STRLEN)
239  return false;
240  uint32 ip = 0;
241  ushort num, shift = 32;
242  const char* p = pstr;
243  for (uint i=0; i < 4; ++i) {
244  num = 0;
245  p = str_scan_decimal(num, p, pend);
246  if (p == NULL)
247  return false;
248  if (i < 3) {
249  if (p == pend || *p != '.')
250  return false;
251  ++p;
252  }
253  if (num > 0xFF)
254  return false;
255  shift -= 8;
256  ip |= (num << shift);
257  }
258  addr_ip4.sin_addr.s_addr = htonl(ip);
259  if (p != pend) {
260  if (*p != ':')
261  return false;
262  num = 0;
263  p = str_scan_decimal(num, p+1, pend);
264  if (p != pend)
265  return false;
266  addr_ip4.sin_port = htons(num);
267  }
268  if (port > 0)
269  addr_ip4.sin_port = htons(port);
270  addrlen = sizeof(struct sockaddr_in);
271  break;
272  }
273  case AF_INET6: {
274  if (str.size() >= MAX_INET6_STRLEN)
275  return false; // too long
276  const char* p = pstr;
277 
278  bool brackets = false;
279  if (*p == '[') {
280  brackets = true;
281  if (++p == pend)
282  return false; // bad syntax
283  }
284 
285  const uint MAX_NUMS = 8;
286  const uint MAX_IPLEN = MAX_NUMS * 2;
287  uint head_len = MAX_IPLEN;
288  uint i = 1;
289  if (*p == ':') {
290  if (++p == pend || *p != ':')
291  return false; // bad syntax
292  if (++p < pend && *p == ']') {
293  if (!brackets)
294  return false; // unexpected ']'
295  i = MAX_NUMS; // stop parse loop
296  }
297  head_len = 0;
298  }
299 
300  // Parsing from first token
301  const char* tokp;
302  uchar* ip = addr_ip6.sin6_addr.s6_addr;
303  uchar* p_ip = ip;
304  ushort num;
305  for (; p < pend && i <= MAX_NUMS; ++i) {
306  if (*p == ':') {
307  if (head_len < MAX_IPLEN)
308  return false; // only 1 "::" allowed
309  head_len = (uint)(p_ip - ip);
310  if (++p == pend)
311  break; // ended with "::"
312  }
313 
314  tokp = p;
315  num = 0;
316  if ((p = str_scan_hex(num, p, pend)) == NULL)
317  return false; // bad num
318 
319  if (p == pend) {
320  if (i < MAX_NUMS && head_len == MAX_IPLEN)
321  return false; // too short, no "::"
322  } else if (*p == '%') {
323  i = MAX_NUMS;
324  } else if (i < MAX_NUMS) {
325  // Not last token, check delim
326  if (*p == '.') {
327  // IPv4 mapped, re-parse from current token
328  p = tokp;
329  if ((uint)(pend - p) > MAX_INET4_STRLEN + 1) // +1 for possible end bracket
330  return false; // too long
331  const uint MAX_IP4LEN = 4;
332  for (uint j=1; j <= MAX_IP4LEN; ++j) {
333  num = 0;
334  if ((p = str_scan_decimal(num, p, pend)) == NULL)
335  return false;
336  if (j < MAX_IP4LEN) {
337  if (p == pend || *p != '.')
338  return false; // too short or bad delim
339  ++p;
340  }
341  if (num > 0xFF)
342  return false; // num out of range
343  *p_ip = (uchar)num;
344  ++p_ip;
345  }
346  break;
347  } else if (*p == ']') {
348  if (!brackets)

350  *p_ip = (uchar)((num >> 8) & 0xFF);
351  *++p_ip = (uchar)(num & 0xFF);
352  ++p_ip;
353  break;
354  } else if (*p != ':')
355  return false; // bad delim
356  ++p;
357  }
358  *p_ip = (uchar)((num >> 8) & 0xFF);
359  *++p_ip = (uchar)(num & 0xFF);
360  ++p_ip;
361  }
362

364  if (head_len < MAX_IPLEN) {

366  if (ip_mov_len >= MAX_IPLEN)
367  return false; // can't use "::" on full IP
368  ip_mov_len -= head_len;
369  if (ip_mov_len > 0)
370  memmove(ip + MAX_IPLEN - ip_mov_len, ip + head_len, ip_mov_len);
371  memset(ip + head_len, 0, MAX_IPLEN - ip_mov_len - head_len);
372  }
373

376  if (++p == pend)
377  return false; // missing scope ID after delim
378  #if !defined(_WIN32)
379  // Linux/Unix: Map interface name to ID
380  if (*p < '0' || *p > '9') {

391  } else

393  if ((p = str_scan_decimal(addr_ip6.sin6_scope_id, p, pend)) == NULL)
394  return false; // bad scope ID
395  }
396 
397  // End bracket
398  if (brackets) {
399  if (p == pend || *p != ']')
400  return false; // missing end ']'
401  ++p;
402  }
403 
404  // Port number
405  if (p < pend) {
406  if (*p != ':' || ++p == pend)
407  return false;
408  num = 0;
409  if ((p = str_scan_decimal(num, p, pend)) != pend)
410  return false;
411  addr_ip6.sin6_port = htons(num);
412  }
413  if (port > 0)
414  addr_ip6.sin6_port = htons(port);

417  break;
418  }

420  return false;
421  }
