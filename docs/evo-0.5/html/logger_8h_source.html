183  assert( msg.size() <= UInt32::MAX );
184 
185  timestamp.set_utc();
186  level = msglevel;
187 
188  const char* p = msg.data();
189  uint32 remain_size = msg.size();
190  if (remain_size > BUF_SIZE) {
191  ::memcpy(buf, p, BUF_SIZE);
192  p += BUF_SIZE;
193  remain_size -= BUF_SIZE;

195  if (aux_buf != NULL) {

202  // New aux_buf

206  copy:

208  ::memcpy(aux_buf, p, remain_size);
209  size = msg.size();
210  } else {
211  if (aux_buf != NULL) {
212  ::free(aux_buf);
213  aux_buf = NULL;

677  const char BEGIN_DELIM = '[';
678  const SubString MSG_DELIM("] ", 2);
679  const char* LEVEL_STR[] = { "ALRT", "ERRR", "WARN", "INFO", "dbug", "dbgl" };
680  const StrSizeT LEVEL_LEN = 4;

683  Logger& logger = *(Logger*)arg;
684  bool closed = false;
685  ulong drop_count = 0;
686 
687  DateTime dt;
688  Msg msg;
689  uint32 buf1_size, buf2_size;
690  for (;;) {
691  // Consume all messages in queue
692  while (logger.queue_.pop(msg)) {
693  if (closed) {
694  ++drop_count;
695  } else {
696  if (logger.local_time_)
697  msg.timestamp.convert_local_dt_notz(dt);
698  else
699  msg.timestamp.convert_utc_dt(dt);
700  dt.tz.set();
701 
702  logger.outfile_ << BEGIN_DELIM;
703  dt.format(logger.outfile_, ':') << ' ';
704  if (msg.level > LOG_LEVEL_DISABLED && msg.level <= LOG_LEVEL_DEBUG_LOW)
705  logger.outfile_ << SubString(LEVEL_STR[(int)msg.level - 1], LEVEL_LEN);
706  else
707  logger.outfile_ << FmtInt(msg.level).width(LEVEL_LEN, ' ');
708 
709  logger.outfile_ << MSG_DELIM;
710 
711  bool error = false;
712  msg.get_buf_sizes(buf1_size, buf2_size);
713  if (buf1_size > 0 && logger.outfile_.writebin(msg.buf, buf1_size) < buf1_size)
714  error = true;
715  else if (buf2_size > 0 && logger.outfile_.writebin(msg.aux_buf, buf2_size) < buf2_size)
716  error = true;
717  else
718  logger.outfile_ << NL;
719  if (error || !logger.outfile_) {
720  // Unable to write to file
721  ++drop_count;
722  logger.outfile_.close();
723  closed = true;
724 
725  Condition::Lock lock(logger.condmutex_);
726  logger.errmsg_.set() << "Logger file write error: " << logger.filepath_ << " -- ";
727  errormsg_out(logger.errmsg_, logger.outfile_.error());
728  break;
729  }
730  }
731  }
732  if (logger.shutdown_.load())
733  break;
734 
735  if (closed || logger.rotate_.load()) {
736  // Log re-open (rotation or log closed from error)
737  {
738  Condition::Lock lock(logger.condmutex_);
739  logger.outfile_.close();
740  if (logger.outfile_.open(logger.filepath_.cstr(), oAPPEND)) {
741  logger.outfile_ << NL;
742  if (closed)
743  logger.outfile_ << "[] Logger recovered from error (lost: " << drop_count << "): " << logger.errmsg_ << NL;
744  logger.errmsg_.set();
745  closed = false;
746  drop_count = 0;
747  } else if (!closed) {
748  logger.errmsg_.set() << "Logger can't open: " << logger.filepath_ << " -- ";
749  errormsg_out(logger.errmsg_, logger.outfile_.error());
750  closed = true;
751  }
752  }
753  logger.rotate_.store(0);
754  }
755 
756  // Wait for more messages -- logging is lock-free (non-blocking) so this must wake up regularly in case a message signal was missed
757  logger.condmutex_.wait(WAKE_TIMEOUT_MS, false);
758  logger.condmutex_.unlock();
759  }
760  logger.shutdown_.store(0);
761  }
