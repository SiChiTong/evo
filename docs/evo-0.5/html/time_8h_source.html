21  inline int parse_num(int& num, const char*& inp, const char* end) {
22  const int OVERFLOW_THRESHOLD = Int::MAX / 10; // not exact, but good enough here
23  const char* start = inp;
24  num = 0;
25  while (inp < end) {
26  if (*inp < '0' || *inp > '9')
27  break;
28  if (num >= OVERFLOW_THRESHOLD) {
29  num = 0;

34  }

900  case 2:
901  break;

911  case 6:

922  return false; // invalid

926  return false;
927  if (inp >= end)
928  break;
929

931  // Fractional hour
932  double frac = impl_time::parse_frac(++inp, end);
933  frac *= MIN_PER_HOUR;
934  minute = (int)frac;
935  frac -= minute;

938  second = (int)frac;
