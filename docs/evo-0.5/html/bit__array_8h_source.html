584  bool format(U& out, int base=fBIN) {
585  if (bitsize_ > 0) {
586  assert( parent_rd_ != NULL );
587  const char* digits;
588  if (base >= 100) {
589  base -= 100;
590  digits = "0123456789abcdefghijklmnopqrstuvwxyz";
591  } else
592  digits = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
593 
594  int bits_per_digit;
595  switch (base) {
596  case 2: bits_per_digit = 1; break;
597  case 4: bits_per_digit = 2; break;
598  case 8: bits_per_digit = 3; break;
599  case 16: bits_per_digit = 4; break;
600  case 32: bits_per_digit = 5; break;
601  default: return false;
602  }
603 
604  typename U::Out& outstream = out.write_out();
605  typename U::Out::Size available = 0;
606 
607  Size outsize = (bitsize_ + bits_per_digit - 1) / bits_per_digit; // round up
608  char* outbuf = outstream.write_direct_multi(available, outsize);
609  if (outbuf == NULL)
610  return false;
611 
612  char* outbuf_start = outbuf;
613  char* outbuf_end = outbuf + available;
614  char* outbuf_write;
615  char* outbuf_next;
616  Size outsize_lastwrite = outsize;
617  Size digit_count;
618 
619  const Value* parent_data = parent_rd_->data_; // local copy of data pointer
620  const Size parent_bitsize = offset_ + bitsize_; // use reduced parent bitsize to exclude ending partial digit bits from extractr()
621 
622  // Use extractr() to extract and format bits
623  const uint digits_per_ulong = IntegerT<ulong>::BITS / bits_per_digit;
624  const uint bits_per_ulong = digits_per_ulong * bits_per_digit;
625  ulong num;
626  outbuf_end -= digits_per_ulong;
627  for (Size p=offset_; outsize > 0; p += bits_per_ulong) {
628  if (outbuf > outbuf_end) {
629  // No room, flush buffer
630  outbuf_start = outbuf = outstream.write_direct_flush(available, (ulong)(outbuf - outbuf_start), outsize);
631  if (outbuf == NULL)
632  return false;
633  outbuf_end = outbuf + available - digits_per_ulong;
634  outsize_lastwrite = outsize;
635  }
636 
637  if (outsize < digits_per_ulong)
638  digit_count = outsize;
639  else
640  digit_count = digits_per_ulong;
641  num = Bits<Value,Size>::template array_extractr<ulong>(parent_data, parent_bitsize, p, digit_count * bits_per_digit);
642 
643  // Format chunk
644  outbuf_next = outbuf_write = outbuf + digit_count;
645  for (; num != 0; num /= (Value)base)
646  *--outbuf_write = digits[num % base];
647  while (outbuf_write > outbuf)
648  *--outbuf_write = '0';
649  outsize -= digit_count;
650  outbuf = outbuf_next;
651  }

665  bool operator==(const ThisType& data) const {
666  bool result;
667  if (this == &data)
668  result = true;
669  else if (parent_rd_ == NULL)
670  result = (data.parent_rd_ == NULL);
671  else if (data.parent_rd_ == NULL || bitsize_ != data.bitsize_)
672  result = false;
673  else if (bitsize_ == 0)
674  result = true;
675  else {
676  const Size this_index = offset_ / Bits<Value,Size>::BITS;
677  const Size this_offset = offset_ - (this_index * Bits<Value,Size>::BITS);
678  const Value* this_ptr = parent_rd_->data_ + this_index;
679  const Size other_index = data.offset_ / Bits<Value,Size>::BITS;
680  const Size other_offset = data.offset_ - (other_index * Bits<Value,Size>::BITS);
681  Size count = bitsize_;
682  Value maskval;
683  if (this_offset == other_offset) {
684  // Easier to compare with same offsets
685  const Value* other_ptr = data.parent_rd_->data_ + other_index;
686 
687  // Leading partial chunk
688  if (this_offset > 0) {
689  maskval = Bits<Value,Size>::ALLBITS >> this_offset;
690  if ((*this_ptr & maskval) != (*other_ptr & maskval))
691  return false;
692  count -= (Bits<Value, Size>::BITS - this_offset);
693  ++this_ptr;
694  ++other_ptr;
695  }
696 
697  // Compare full chunks
698  while (count >= Bits<Value, Size>::BITS) {
699  count -= Bits<Value, Size>::BITS;
700  if (*this_ptr != *other_ptr)
701  return false;
702  ++this_ptr;
703  ++other_ptr;
704  }
705 
706  // Compare trailing partial chunk
707  if (count > 0) {
708  maskval = ~(Bits<Value,Size>::ALLBITS >> count);
709  if ((*this_ptr & maskval) != (*other_ptr & maskval))
710  return false;
711  }
712  } else {
713  // Trickier to compare with different offsets, use array_extractl() for other data
714  const Value* other_ptr = data.parent_rd_->data_;
715  const Size other_bitsize = data.parent_rd_->bitsize_;
716  Size other_pos = data.offset_;
717 
718  // Leading partial chunk
719  if (this_offset > 0) {
720  maskval = Bits<Value,Size>::ALLBITS >> this_offset;
721  if ( (*this_ptr & maskval) != Bits<Value,Size>::template array_extractr<Value>(other_ptr, other_bitsize, other_pos, (Bits<Value,Size>::BITS - this_offset)) )
722  return false;
723  ++this_ptr;
724  const Size bits = Bits<Value, Size>::BITS - this_offset;
725  count -= bits;
726  other_pos += bits;
727  }
728 
729  // Compare full chunks
730  while (count >= Bits<Value, Size>::BITS) {
731  count -= Bits<Value, Size>::BITS;
732  if (*this_ptr != Bits<Value,Size>::template array_extractl<Value>(other_ptr, other_bitsize, other_pos, Bits<Value,Size>::BITS))
733  return false;
734  ++this_ptr;
735  other_pos += Bits<Value, Size>::BITS;
736  }
737 
738  // Compare trailing partial chunk
739  if (count > 0) {
740  maskval = ~(Bits<Value,Size>::ALLBITS >> count);
741  if ((*this_ptr & maskval) != Bits<Value,Size>::template array_extractl<Value>(other_ptr, other_bitsize, other_pos, count))
742  return false;
743  }
744  }
745  result = true;
746  }
747  return result;
748  }

1416  typedef typename StaticIf<(sizeof(T) > sizeof(ulong)), T, ulong>::Type TNum;
1417 
1418  int bits_per_digit;
1419  switch (base) {
1420  case 2: bits_per_digit = 1; break;
1421  case 4: bits_per_digit = 2; break;
1422  case 8: bits_per_digit = 3; break;
1423  case 16: bits_per_digit = 4; break;
1424  case 32: bits_per_digit = 5; break;
1425  default: return 0;
1426  }
1427 
1428  uchar ch;
1429  while (size > 0 && ((ch=str[size-1]) == ' ' || ch == '\t'))
1430  --size;
1431  const char* end = str + size;
1432  while (str < end && ((ch=*str) == ' ' || ch == '\t'))
1433  ++str;
1434  if (str >= end)
1435  return 0;
1436 
1437  size = (Size)(end - str) * bits_per_digit; // size is now in bits
1438  resize(size);
1439 
1440  uint bits = 0;
1441  TNum num = 0;
1442  T* p = data_;
1443  for (; str < end; ++str) {
1444  ch = *str;
1445  if (ch >= '0' && ch <= '9')
1446  ch -= '0';
1447  else if (ch >= 'A' && ch <= 'V')
1448  ch = ch - 'A' + 10;
1449  else if (ch >= 'a' && ch <= 'v')
1450  ch = ch - 'a' + 10;
1451  else
1452  return 0; // not a digit
1453  if (ch >= base)
1454  return 0;
1455 
1456  bits += bits_per_digit;
1457  if (bits == Bits<T, Size>::BITS) {
1458  // No overflow
1459  *p = T((num * (uint)base) + ch);
1460  ++p;
1461  num = 0;
1462  bits = 0;
1463  } else if (bits > Bits<T,Size>::BITS) {
1464  // Protect from overflow from extra bits
1465  const uint shift = Bits<T,Size>::BITS - (bits - bits_per_digit);
1466  bits -= Bits<T,Size>::BITS;
1467  *p = T(num << shift) | (T(ch) >> bits);
1468  ++p;
1469  num = T(ch) & ~(Bits<T,Size>::ALLBITS << bits);
1470  } else {
1471  num *= (uint)base;
1472  num += ch;
1473  }
1474  }
1475

1477  *p = T(num) << (Bits<T,Size>::BITS - bits);
1478  return size;

1511  case 32: bits_per_digit = 5; break;

1520  if (outbuf == NULL)

1526  char* outbuf_next;

1531  const uint digits_per_ulong = IntegerT<ulong>::BITS / bits_per_digit;

1543  outbuf_end = outbuf + available - digits_per_ulong;
1544  outsize_lastwrite = outsize;
1545  }

1551  while (outbuf_write > outbuf)
1552  *--outbuf_write = '0';
1553  outsize -= digit_count;
1554  outbuf = outbuf_next;
1555  }

1579  }

1586  num >>= Bits<T,Size>::BITS - (outsize * bits_per_digit);

1590  }
