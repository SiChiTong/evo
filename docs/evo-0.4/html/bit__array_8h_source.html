641  bool result;

643  result = true;
644  else if (parent_rd_ == NULL)
645  result = (data.parent_rd_ == NULL);

647  result = false;
648  else if (bitsize_ == 0)
649  result = true;

652  const Size this_offset = offset_ - (this_index * Bits<Value,Size>::BITS);
653  const Value* this_ptr = parent_rd_->data_ + this_index;

655  const Size other_offset = data.offset_ - (other_index * Bits<Value,Size>::BITS);
656  Size count = bitsize_;

658  if (this_offset == other_offset) {
659  // Easier to compare with same offsets
660  const Value* other_ptr = data.parent_rd_->data_ + other_index;
661 
662  // Leading partial chunk
663  if (this_offset > 0) {
664  maskval = Bits<Value,Size>::ALLBITS >> this_offset;
665  if ((*this_ptr & maskval) != (*other_ptr & maskval))
666  return false;
667  count -= (Bits<Value, Size>::BITS - this_offset);
668  ++this_ptr;
669  ++other_ptr;
670  }
671 
672  // Compare full chunks
673  while (count >= Bits<Value, Size>::BITS) {
674  count -= Bits<Value, Size>::BITS;
675  if (*this_ptr != *other_ptr)
676  return false;
677  ++this_ptr;
678  ++other_ptr;
679  }
680 
681  // Compare trailing partial chunk
682  if (count > 0) {
683  maskval = ~(Bits<Value,Size>::ALLBITS >> count);
684  if ((*this_ptr & maskval) != (*other_ptr & maskval))
685  return false;
686  }
687  } else {

696  if ( (*this_ptr & maskval) != Bits<Value,Size>::template array_extractr<Value>(other_ptr, other_bitsize, other_pos, (Bits<Value,Size>::BITS - this_offset)) )

699  const Size bits = Bits<Value, Size>::BITS - this_offset;
700  count -= bits;
701  other_pos += bits;
702  }
703

705  while (count >= Bits<Value, Size>::BITS) {
706  count -= Bits<Value, Size>::BITS;
707  if (*this_ptr != Bits<Value,Size>::template array_extractl<Value>(other_ptr, other_bitsize, other_pos, Bits<Value,Size>::BITS))

716  if ((*this_ptr & maskval) != Bits<Value,Size>::template array_extractl<Value>(other_ptr, other_bitsize, other_pos, count))
